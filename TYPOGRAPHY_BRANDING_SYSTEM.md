from PIL import Image, ImageDraw, ImageFont
import numpy as np

def create_saymurr_intro_card(duration=3.5):
    """
    Intro card: –°–ê–ô–ú–£–†–† —Å —Å–∏–º–≤–æ–ª–æ–º –≥–ª–∞–∑–∞
    0.5-4.0s
    """
    def make_frame(t):
        # Background
        img = Image.new('RGB', (1920, 1080), (12, 12, 12))
        draw = ImageDraw.Draw(img)

        # Timing phases
        if t < 0.5:
            # Pure black
            pass

        elif t < 1.5:
            # Symbol ‚óâ animation
            progress = (t - 0.5) / 1.0
            alpha = min(1.0, progress * 1.5)  # Fade in

            # Blink animation
            if 0.6 < progress < 0.7:
                scale = 1.0 + (progress - 0.6) * 1.5  # Expand
            elif 0.7 < progress < 0.8:
                scale = 1.15 - (progress - 0.7) * 1.5  # Contract
            else:
                scale = 1.0

            # Draw eye symbol
            eye_color = (int(48*alpha), int(168*alpha), int(107*alpha))
            eye_x, eye_y = 960, 500
            eye_radius = int(12 * scale)

            draw.ellipse(
                [eye_x - eye_radius, eye_y - eye_radius,
                 eye_x + eye_radius, eye_y + eye_radius],
                fill=eye_color
            )

            # Inner pupil
            pupil_radius = int(5 * scale)
            draw.ellipse(
                [eye_x - pupil_radius, eye_y - pupil_radius,
                 eye_x + pupil_radius, eye_y + pupil_radius],
                fill=(12, 12, 12)
            )

        elif t < 4.0:
            # Text –°–ê–ô–ú–£–†–†
            progress = (t - 1.5) / 0.5  # 0‚Üí1 during 1.5-2.0s

            if t < 2.0:
                # Fade in
                alpha = min(1.0, progress)
                y_offset = int(20 * (1 - alpha))
            elif t < 3.5:
                # Hold
                alpha = 1.0
                y_offset = 0
            else:
                # Fade out
                fade_progress = (t - 3.5) / 0.5
                alpha = 1.0 - fade_progress
                y_offset = 0

            # Load font
            try:
                font = ImageFont.truetype(
                    "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf",
                    140
                )
                small_font = ImageFont.truetype(
                    "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf",
                    20
                )
            except:
                font = ImageFont.load_default()
                small_font = ImageFont.load_default()

            # Draw "–°" separately (green)
            green = (int(48*alpha), int(168*alpha), int(107*alpha))
            white = (int(252*alpha), int(252*alpha), int(250*alpha))
            gray = (int(139*alpha), int(139*alpha), int(142*alpha))

            # Calculate positions with wide tracking
            text_y = 480 + y_offset

            # "–°" (first letter, green, with glow)
            c_x = 680
            draw.text((c_x, text_y), "–°", fill=green, font=font)

            # Rest of letters with spacing
            letters = " –ê –ô –ú –£ –† –†"
            letter_spacing = 52  # Wide tracking
            x = c_x + 90  # Start after "–°"

            for letter in letters:
                if letter != " ":
                    draw.text((x, text_y), letter, fill=white, font=font)
                x += letter_spacing

            # Transliteration below
            translit = "s a y m u r r"
            translit_bbox = draw.textbbox((0, 0), translit, font=small_font)
            translit_width = translit_bbox[2] - translit_bbox[0]
            translit_x = (1920 - translit_width) // 2
            draw.text((translit_x, text_y + 160), translit, fill=gray, font=small_font)

        # Convert to numpy
        frame = np.array(img)

        # Add film grain
        grain = np.random.normal(0, 12, frame.shape)
        frame = np.clip(frame.astype(float) + grain, 0, 255).astype(np.uint8)

        return frame

    return VideoClip(make_frame, duration=duration)


def create_saymurr_outro_card(duration=1.5):
    """
    Outro card: Full info
    35.5-37.0s
    """
    def make_frame(t):
        img = Image.new('RGB', (1920, 1080), (12, 12, 12))
        draw = ImageDraw.Draw(img)

        # Timing
        if t < 0.3:
            # –°–ê–ô–ú–£–†–† fade in
            alpha = t / 0.3
            show_artist = True
            show_track = False
            show_subtitle = False
        elif t < 0.6:
            # Add –°–ú–û–¢–†–ï–õ–ê
            alpha = 1.0
            show_artist = True
            show_track = True
            show_subtitle = False
        elif t < 1.0:
            # Add look at me
            alpha = 1.0
            show_artist = True
            show_track = True
            show_subtitle = True
        else:
            # Hold with pulse
            alpha = 1.0
            show_artist = True
            show_track = True
            show_subtitle = True

            # Pulse calculation
            pulse_progress = (t - 1.0) / 0.5
            if int(pulse_progress * 2) % 2 == 0:
                pulse_intensity = abs(np.sin(pulse_progress * 2 * np.pi))
            else:
                pulse_intensity = 0

        # Colors
        white = (int(252*alpha), int(252*alpha), int(250*alpha))
        gray = (int(139*alpha*0.55), int(139*alpha*0.55), int(142*alpha*0.55))
        green = (int(48*alpha), int(168*alpha), int(107*alpha))

        # Fonts
        try:
            artist_font = ImageFont.truetype(
                "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf",
                140
            )
            track_font = ImageFont.truetype(
                "/usr/share/fonts/truetype/dejavu/DejaVuSans-Bold.ttf",
                88
            )
            subtitle_font = ImageFont.truetype(
                "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf",
                26
            )
            small_font = ImageFont.truetype(
                "/usr/share/fonts/truetype/dejavu/DejaVuSans.ttf",
                18
            )
        except:
            artist_font = track_font = subtitle_font = small_font = ImageFont.load_default()

        base_y = 360

        # Artist name
        if show_artist:
            # "–°" with potential glow
            c_x = 580
            c_y = base_y

            if t > 1.0:
                # Add glow
                glow_alpha = int(pulse_intensity * 100)
                for radius in [60, 40, 20]:
                    glow_color = (48, 168, 107, glow_alpha)
                    # Note: PIL doesn't support RGBA glow easily,
                    # would need to composite multiple layers

            draw.text((c_x, c_y), "–°", fill=green, font=artist_font)

            # Rest
            letters = " –ê –ô –ú –£ –† –†"
            x = c_x + 90
            for letter in letters:
                if letter != " ":
                    draw.text((x, c_y), letter, fill=white, font=artist_font)
                x += 52

            # Transliteration
            translit = "s a y m u r r"
            translit_bbox = draw.textbbox((0, 0), translit, font=small_font)
            translit_x = (1920 - translit_bbox[2]) // 2
            draw.text((translit_x, c_y + 155), translit, fill=gray, font=small_font)

        # Separator line
        if show_track:
            line_y = base_y + 210
            line_x1 = 860
            line_x2 = 1060
            draw.line([line_x1, line_y, line_x2, line_y], fill=green, width=2)

        # Track name
        if show_track:
            track_text = "–°–ú–û–¢–†–ï–õ–ê"
            track_bbox = draw.textbbox((0, 0), track_text, font=track_font)
            track_width = track_bbox[2] - track_bbox[0]
            track_x = (1920 - track_width) // 2
            track_y = base_y + 240
            draw.text((track_x, track_y), track_text, fill=white, font=track_font)

        # Subtitle
        if show_subtitle:
            sub_text = "look at me"
            sub_bbox = draw.textbbox((0, 0), sub_text, font=subtitle_font)
            sub_width = sub_bbox[2] - sub_bbox[0]
            sub_x = (1920 - sub_width) // 2
            sub_y = base_y + 350
            draw.text((sub_x, sub_y), sub_text, fill=gray, font=subtitle_font)

        frame = np.array(img)
        frame = add_film_grain(frame, 0.12)

        return frame

    return VideoClip(make_frame, duration=duration)

def add_film_grain(frame, intensity=0.12):
    """Add cinematic grain"""
    grain = np.random.normal(0, intensity * 255, frame.shape)
    result = np.clip(frame.astype(float) + grain, 0, 255)
    return result.astype(np.uint8)
